#include <iostream>

using namespace std;

int main() //return type
{
    char statement[100];
    int op1, op2;
    char operation;
    char answer = 'Y'; //char must be in single quote and end with semicolon
    while (answer == 'Y' || answer == 'y') //must be here given char in single quote and
    {
        cout << "Enter expression" << endl;
        cin >> op2 >> operation >> op1;
        if (operation == '+') //operator must be a character in single quote and not the semicolon
            cout << op1 << " + " << op2 << " = " << op1 + op2 << endl; //must be all out operator
        if (operation == '-')
            cout << op1 << " - " << op2 << " = " << op1 - op2 << endl; //must be all out operator
        if (operation == '*')
            cout << op1 << " / " << op2 << " = " << op1 * op2 << endl; //must be semicolon in statement end
        if (operation == '/')
            cout << op1 << " * " << op2 << " = " << op1 / op2 << endl;

        cout << "Do you wish to evaluate another expression? " << endl;
        cin >> answer;
    }
    if (answer == 'N' || answer == 'n')
        cout << "Program Finished";
}
